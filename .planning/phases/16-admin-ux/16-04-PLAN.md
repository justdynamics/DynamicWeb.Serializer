    ```csharp
    namespace DynamicWeb.Serializer.AdminUI.Injectors;

    using Dynamicweb.CoreUI;
    using Dynamicweb.CoreUI.Actions;
    using Dynamicweb.CoreUI.Actions.Implementations;
    using Dynamicweb.CoreUI.Icons;
    using Dynamicweb.CoreUI.Screens;
    using Dynamicweb.Files.UI.Screens.Files;
    using DynamicWeb.Serializer.Configuration;

    public sealed class SerializerFileOverviewInjector : ScreenInjector<FileOverviewScreen>
    {
        public override void OnAfter(FileOverviewScreen screen, UiComponentBase content)
        {
            var model = Screen?.Model;
            if (model is null)
                return;

            if (!string.Equals(model.Extension, ".zip", StringComparison.OrdinalIgnoreCase))
                return;

            if (!IsInOutputDirectory(model.FilePath))
                return;

            content.TryGet<ScreenLayout>(out var screenLayout);
            screenLayout?.ContextActionGroups.Add(new ActionGroup
            {
                Nodes = [
                    new ActionNode
                    {
                        Name = "Import to database",
                        Icon = Icon.Upload,
                        NodeAction = ExecuteCommandAction.Using(
                            new DeserializeFromZipCommand { FilePath = model.FilePath })
                    }
                ]
            });
        }

        internal static bool IsInOutputDirectory(string? filePath)
        {
            if (string.IsNullOrEmpty(filePath))
                return false;

            var configPath = ConfigPathResolver.FindConfigFile();
            if (configPath == null)
                return false;

            var config = ConfigLoader.Load(configPath);
            var outputDir = config.OutputDirectory?.Replace('\\', '/').TrimEnd('/') ?? "";
            var normalizedPath = filePath.Replace('\\', '/');

            return normalizedPath.Contains(outputDir, StringComparison.OrdinalIgnoreCase);
        }
    }
    ```

  <action>
    **Create `DeserializeFromZipCommand.cs`:**
    A `CommandBase` that receives `FilePath` (the zip file path), executes deserialization:

    1. Load config via `ConfigPathResolver.FindConfigFile()` + `ConfigLoader.Load()`
    2. Validate the zip file exists and is a .zip
    3. Extract zip to a temp directory via `ZipFile.ExtractToDirectory()` (using `System.IO.Compression`)
    4. Validate the extracted content has YAML files matching configured predicates (per D-21). If not, clean up temp dir and return error: "This zip doesn't contain valid serialization data. Expected YAML files matching configured predicates."
    5. Create orchestrator via `ProviderRegistry.CreateOrchestrator(filesRoot)`
    6. Run dry-run first: `orchestrator.DeserializeAll(config.Predicates, tempDir, log, isDryRun: true)`
    7. If dry-run has errors, report them and clean up
    8. Run actual deserialization: `orchestrator.DeserializeAll(config.Predicates, tempDir, log, isDryRun: false)`
    9. Build `LogFileSummary` from result, generate advice via `AdviceGenerator.GenerateAdvice(result)`
    10. Write per-run log file via `LogFileWriter.CreateLogFile(paths.Log, "DeserializeZip")` + `LogFileWriter.WriteSummaryHeader()` + flush log lines
    11. Clean up temp directory in a finally block (per D-20)
    12. Return `CommandResult` with summary message

    Note on the D-19 flow (dry-run -> confirm -> execute): The simplest implementation is to have the command do both dry-run and execute in sequence, reporting the combined result. A two-step dialog flow would require a more complex screen setup. If the DW admin framework supports a confirmation dialog pattern, use it. Otherwise, the command can execute directly (the dry-run result is included in the log file for reference).
