    ```powershell
    <#
    .SYNOPSIS
    Post-deserialize frontend smoke test for a DynamicWeb Swift baseline deploy.

    .DESCRIPTION
    Enumerates every `PageActive = 1` page under the target area in the DW DB,
    hits the public Friendly URL for each, buckets responses into 2xx/3xx/4xx/5xx,
    and reports. Exits non-zero on any 5xx. Designed to be the last step of a
    round-trip verification.

    Phase 38 D-38-13. LOCAL-DEV ONLY — never deploy to customer sites.

    .PARAMETER HostUrl
    Public hostname of the DW target. Default https://localhost:58217 (CleanDB).

    .PARAMETER ConnectionString
    SQL Server connection string for enumeration. Default matches CleanDB.

    .PARAMETER AreaId
    The PageAreaID to enumerate. Default 3 (Swift 2 on CleanDB).

    .PARAMETER LangPrefix
    Friendly URL language prefix. Default /en-us. Use empty string to disable.

    .EXAMPLE
    pwsh tools/smoke/Test-BaselineFrontend.ps1

    .EXAMPLE
    pwsh tools/smoke/Test-BaselineFrontend.ps1 -HostUrl https://myhost:58217 -AreaId 3 -LangPrefix /en-us
    #>
    param(
        [string]$HostUrl = 'https://localhost:58217',
        [string]$ConnectionString = 'Server=localhost\SQLEXPRESS;Database=Swift-CleanDB;Integrated Security=true;TrustServerCertificate=true',
        [int]$AreaId = 3,
        [string]$LangPrefix = '/en-us'
    )

    Write-Host "=== DynamicWeb.Serializer Frontend Smoke Test ===" -ForegroundColor Cyan
    Write-Host "Host:     $HostUrl"
    Write-Host "AreaID:   $AreaId"
    Write-Host "LangPref: $LangPrefix"
    Write-Host ""

    # --- Enumerate active pages ---------------------------------------------
    $pageQuery = @"
    SELECT PageID, PageMenuText, PageUrlName
    FROM Page
    WHERE PageAreaID = $AreaId AND PageActive = 1 AND (PageDeleted = 0 OR PageDeleted IS NULL)
    ORDER BY PageID;
    "@

    try {
        $rows = Invoke-Sqlcmd -ConnectionString $ConnectionString -Query $pageQuery -ErrorAction Stop
    } catch {
        Write-Host "ERROR: Could not enumerate pages — $_" -ForegroundColor Red
        Write-Host "Fallback: install SqlServer module via 'Install-Module -Name SqlServer'" -ForegroundColor Yellow
        exit 2
    }

    if (-not $rows) {
        Write-Host "No active pages found under area $AreaId. Nothing to test." -ForegroundColor Yellow
        exit 0
    }

    Write-Host "Found $($rows.Count) active page(s) to smoke-test." -ForegroundColor Green
    Write-Host ""

    # --- Hit each and bucket -----------------------------------------------
    $buckets = @{ '2xx' = @(); '3xx' = @(); '4xx' = @(); '5xx' = @() }
    $errors = @()

    foreach ($r in $rows) {
        $slug = $r.PageUrlName
        if ([string]::IsNullOrWhiteSpace($slug)) {
            $url = "$HostUrl/Default.aspx?ID=$($r.PageID)"
        } else {
            $url = "$HostUrl$LangPrefix/$slug"
        }

        $code = 0
        $body = $null
        $headers = $null
        $finalUrl = $url

        try {
            $resp = Invoke-WebRequest -Uri $url `
                -SkipCertificateCheck `
                -MaximumRedirection 5 `
                -ErrorAction Stop `
                -TimeoutSec 30
            $code = [int]$resp.StatusCode
            $body = $resp.Content
            $headers = $resp.Headers
            if ($resp.BaseResponse -and $resp.BaseResponse.RequestMessage -and $resp.BaseResponse.RequestMessage.RequestUri) {
                $finalUrl = $resp.BaseResponse.RequestMessage.RequestUri.AbsoluteUri
            }
        } catch {
            # Non-2xx responses throw in PS7; extract status from the exception.
            if ($_.Exception.Response) {
                $code = [int]$_.Exception.Response.StatusCode
                try { $body = (New-Object System.IO.StreamReader($_.Exception.Response.GetResponseStream())).ReadToEnd() } catch { $body = "" }
                try { $headers = $_.Exception.Response.Headers } catch { $headers = $null }
            } else {
                $errors += [PSCustomObject]@{ PageId = $r.PageID; Url = $url; Error = $_.Exception.Message }
                Write-Host "  [$($r.PageID)] $url -> TRANSPORT ERROR: $($_.Exception.Message)" -ForegroundColor Red
                continue
            }
        }

        $bucket = if ($code -ge 500) { '5xx' }
                  elseif ($code -ge 400) { '4xx' }
                  elseif ($code -ge 300) { '3xx' }
                  elseif ($code -ge 200) { '2xx' }
                  else { '5xx' }  # unexpected; treat as failure

        $entry = [PSCustomObject]@{
            PageId = $r.PageID
            MenuText = $r.PageMenuText
            Url = $url
            FinalUrl = $finalUrl
            Code = $code
        }
        if ($bucket -eq '5xx' -and $body) {
            $entry | Add-Member -NotePropertyName BodyExcerpt -NotePropertyValue ($body.Substring(0, [Math]::Min(2000, $body.Length)))
            $entry | Add-Member -NotePropertyName Headers -NotePropertyValue $headers
        } elseif ($bucket -eq '4xx' -and $body) {
            $entry | Add-Member -NotePropertyName BodyExcerpt -NotePropertyValue ($body.Substring(0, [Math]::Min(500, $body.Length)))
        }

        $buckets[$bucket] += $entry
        $color = switch ($bucket) { '2xx' {'Green'} '3xx' {'Cyan'} '4xx' {'Yellow'} default {'Red'} }
        Write-Host "  [$($r.PageID)] $url -> $code ($bucket)" -ForegroundColor $color
    }

    # --- Summary ------------------------------------------------------------
    Write-Host ""
    Write-Host "=== Summary ===" -ForegroundColor Cyan
    Write-Host "2xx: $($buckets['2xx'].Count)"
    Write-Host "3xx: $($buckets['3xx'].Count)"
    Write-Host "4xx: $($buckets['4xx'].Count)"
    Write-Host "5xx: $($buckets['5xx'].Count)"
    Write-Host "Transport errors: $($errors.Count)"

    if ($buckets['5xx'].Count -gt 0) {
        Write-Host ""
        Write-Host "=== 5xx Details ===" -ForegroundColor Red
        $buckets['5xx'] | ConvertTo-Json -Depth 4 | Write-Host
    }
    if ($buckets['4xx'].Count -gt 0) {
        Write-Host ""
        Write-Host "=== 4xx Details (first 500 chars) ===" -ForegroundColor Yellow
        $buckets['4xx'] | Select-Object PageId, Url, Code, BodyExcerpt | Format-Table -AutoSize
    }

    if ($buckets['5xx'].Count -gt 0 -or $errors.Count -gt 0) {
        exit 1
    }
    exit 0
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Local dev machine → SQL Server on localhost | Tool uses Integrated Security — runs as the current user's Windows identity. |
| Local dev machine → DW host over HTTPS | Tool makes GET requests to public pages. `-SkipCertificateCheck` bypasses self-signed cert validation (localhost-only). |
| Tool → user terminal | Tool prints page URLs, status codes, and 4xx/5xx body excerpts. No sensitive output (no auth tokens, no DB connection strings shown). |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-38-D3-01 | Tampering | Runaway redirect chain to arbitrary host | mitigate | `-MaximumRedirection 5` caps redirect chase. `$HostUrl` is a local parameter, NOT sourced from env or config. |
| T-38-D3-02 | Information Disclosure | Body excerpts in terminal output | accept | Local-dev only per D-38-13. Body content from DW pages is not more sensitive than what the local dev has access to via browser. |
| T-38-D3-03 | Elevation of Privilege | Tool running against production | mitigate | README prominently says `LOCAL-DEV ONLY — never deploy to customer sites`. Defaults are localhost-only. |
| T-38-D3-04 | Denial of Service | Unbounded page enumeration hammering target | mitigate | Sequential requests (not parallel). `-TimeoutSec 30` per page. 80-page baseline caps total runtime to ~40 min worst case. |

**Not applicable:** No cross-process auth, no secret handling, no mutable state on the target.
</threat_model>
