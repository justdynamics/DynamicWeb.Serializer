**2. Add ReadAreaProperties helper** to ContentMapper:

Add a static method `ReadAreaProperties(int areaId, IReadOnlySet<string>? excludeFields)` that reads all columns from the [Area] SQL table via direct SQL (the DW Area C# class does NOT expose all 60+ columns as properties, so direct SQL is the only way per architecture research).

Use `Dynamicweb.Data.Database.CreateDataReader` directly (NOT ISqlExecutor -- that abstraction lives in SqlTable namespace and ContentMapper is in Serialization namespace; using Database static directly is simpler and consistent with how ContentDeserializer will also need direct SQL access).

```csharp
using Dynamicweb.Data;

private static Dictionary<string, object> ReadAreaProperties(int areaId, IReadOnlySet<string>? excludeFields)
{
    var props = new Dictionary<string, object>(StringComparer.OrdinalIgnoreCase);
    var cb = new CommandBuilder();
    cb.Add("SELECT * FROM [Area] WHERE [AreaID] = @id");
    cb.AddParameter("id", areaId);
    using var reader = Database.CreateDataReader(cb);
    if (reader.Read())
    {
        for (int i = 0; i < reader.FieldCount; i++)
        {
            var name = reader.GetName(i);
            var value = reader.GetValue(i);
            if (value != DBNull.Value && excludeFields?.Contains(name) != true)
                props[name] = value;
        }
    }
    // Remove columns already captured by named DTO properties to avoid duplication
    props.Remove("AreaID");
    props.Remove("AreaName");
    props.Remove("AreaSort");
    props.Remove("AreaItemType");
    props.Remove("AreaItemId");
    props.Remove("AreaUniqueId");
    return props;
}
```

The columns removed (AreaID, AreaName, AreaSort, AreaItemType, AreaItemId, AreaUniqueId) are already represented by the named properties on SerializedArea (AreaId=UniqueId GUID, Name=AreaName, SortOrder=AreaSort, ItemType=AreaItemType). Keeping them in Properties would cause duplication in area.yml.

**2. Add WriteAreaProperties helper method** to ContentDeserializer:

```csharp
/// <summary>
/// Write area properties back to the [Area] table via SQL UPDATE.
/// Skips columns in excludeFields to preserve environment-specific values.
/// </summary>
private void WriteAreaProperties(int areaId, Dictionary<string, object> properties, IReadOnlySet<string>? excludeFields)
{
    if (properties.Count == 0) return;

    var cb = new CommandBuilder();
    cb.Add("UPDATE [Area] SET ");
    var first = true;
    foreach (var kvp in properties)
    {
        // Skip excluded fields (per AREA-05)
        if (excludeFields?.Contains(kvp.Key) == true) continue;

        if (!first) cb.Add(", ");
        cb.Add($"[{kvp.Key}] = @p_{kvp.Key}");
        cb.AddParameter($"p_{kvp.Key}", kvp.Value ?? DBNull.Value);
        first = false;
    }
    // If all properties were excluded, nothing to update
    if (first) return;

    cb.Add(" WHERE [AreaID] = @areaId");
    cb.AddParameter("areaId", areaId);
    Database.ExecuteNonQuery(cb);
}
```

Use `@p_` prefix on parameters to avoid collisions with SQL reserved words. The `first` flag check after the loop handles the edge case where excludeFields eliminates all properties.

**3. Add CreateAreaFromProperties helper method** to ContentDeserializer:

```csharp
/// <summary>
/// Create a new area row via SQL INSERT using serialized properties.
/// Called when the target area does not exist (AREA-04).
/// </summary>
private void CreateAreaFromProperties(int areaId, SerializedArea area, IReadOnlySet<string>? excludeFields)
{
    var cb = new CommandBuilder();
    var columns = new List<string> { "[AreaID]", "[AreaName]", "[AreaSort]", "[AreaUniqueId]" };
    var paramNames = new List<string> { "@p_AreaID", "@p_AreaName", "@p_AreaSort", "@p_AreaUniqueId" };
    cb.AddParameter("p_AreaID", areaId);
    cb.AddParameter("p_AreaName", area.Name);
    cb.AddParameter("p_AreaSort", area.SortOrder);
    cb.AddParameter("p_AreaUniqueId", area.AreaId); // AreaId on DTO is the GUID UniqueId

    foreach (var kvp in area.Properties)
    {
        if (excludeFields?.Contains(kvp.Key) == true) continue;
        columns.Add($"[{kvp.Key}]");
        paramNames.Add($"@p_{kvp.Key}");
        cb.AddParameter($"p_{kvp.Key}", kvp.Value ?? DBNull.Value);
    }

    cb.Add($"INSERT INTO [Area] ({string.Join(", ", columns)}) VALUES ({string.Join(", ", paramNames)})");
    Database.ExecuteNonQuery(cb);
}
```
