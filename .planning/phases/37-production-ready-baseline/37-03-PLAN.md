<objective>
Three narrow but high-leverage safety improvements, batched because they share the same attack surface (SQL identifiers in config) and the same config-load validation choke point:

1. **SqlTable `where` clause (FILTER-01)** — Let a predicate target `AccessUser WHERE AccessUserType = 2 AND AccessUserUserName IN ('Admin','Editors')` so the baseline can include admin/editor role rows without dragging in tenant demo user groups. Closes F-02's headline "all-or-nothing" limitation on SqlTable predicates.

2. **SQL identifier whitelist (SEED-002)** — Validate every table name, column name, and Where-clause identifier against INFORMATION_SCHEMA at config load time. Today identifiers are spliced into SQL text unvalidated (parameterization doesn't apply to identifiers). Low-risk today (trusted admin-only config), but SEED-002 flagged it as an escalating risk once configs come from less-trusted sources.

3. **Runtime-column auto-exclusion (RUNTIME-COLS-01)** — A curated flat list of runtime-only columns (UrlPathVisitsCount, EcomShops.ShopIndex*, etc.) that are silently excluded from serialization even if the predicate doesn't list them. Single-category list per D-07 — NO Runtime-vs-Credential classification.

**Out of scope:**
- Credential columns registry (CRED-01) — DEFERRED to v0.6.0 per D-09 (see DEFERRED.md)
- SQL injection in the deserialize MERGE path — already parameterized (CommandBuilder)
- Rewriting existing baselines that may contain leaked credentials — documentation-only remediation in README
</objective>

    ```csharp
    using Dynamicweb.Data;
    namespace DynamicWeb.Serializer.Configuration;

    /// <summary>
    /// Validates SQL identifiers (table / column names) against INFORMATION_SCHEMA.
    /// Identifiers cannot be parameterized in T-SQL (they're spliced as text), so the
    /// only defense against "'; DROP TABLE X;--" as a table name is allowlisting.
    /// Per SEED-002.
    /// </summary>
    public class SqlIdentifierValidator
    {
        private readonly Dictionary<string, HashSet<string>> _tableColumns =
            new(StringComparer.OrdinalIgnoreCase);
        private HashSet<string>? _tableNames;

        /// <summary>
        /// Validate a table name exists in INFORMATION_SCHEMA.TABLES. Throws on mismatch.
        /// </summary>
        public void ValidateTable(string tableName)
        {
            if (string.IsNullOrWhiteSpace(tableName))
                throw new InvalidOperationException("Empty table name is not a valid identifier.");

            EnsureTableNames();
            if (!_tableNames!.Contains(tableName))
                throw new InvalidOperationException(
                    $"Table identifier not in INFORMATION_SCHEMA: '{tableName}'. " +
                    "Check the 'table' value in your predicate config.");
        }

        public void ValidateColumn(string tableName, string columnName)
        {
            if (string.IsNullOrWhiteSpace(columnName))
                throw new InvalidOperationException("Empty column name is not a valid identifier.");

            // Only call after table is validated — otherwise skip (caller's table check will surface)
            if (!_tableColumns.ContainsKey(tableName)) LoadColumns(tableName);
            var cols = _tableColumns[tableName];
            if (!cols.Contains(columnName))
                throw new InvalidOperationException(
                    $"Column identifier not in INFORMATION_SCHEMA: '[{tableName}].[{columnName}]'. " +
                    "Check exclude/include/where fields in your predicate config.");
        }

        /// <summary>Returns the cached column set for a table. Loads from INFORMATION_SCHEMA on first call.</summary>
        public HashSet<string> GetColumns(string tableName)
        {
            if (!_tableColumns.ContainsKey(tableName)) LoadColumns(tableName);
            return _tableColumns[tableName];
        }

        private void EnsureTableNames()
        {
            if (_tableNames != null) return;
            var set = new HashSet<string>(StringComparer.OrdinalIgnoreCase);
            var cb = new CommandBuilder();
            cb.Add("SELECT TABLE_NAME FROM INFORMATION_SCHEMA.TABLES WHERE TABLE_TYPE = 'BASE TABLE'");
            using var reader = Database.CreateDataReader(cb);
            while (reader.Read()) set.Add(reader.GetString(0));
            _tableNames = set;
        }

        private void LoadColumns(string tableName)
        {
            var cols = new HashSet<string>(StringComparer.OrdinalIgnoreCase);
            var cb = new CommandBuilder();
            cb.Add("SELECT COLUMN_NAME FROM INFORMATION_SCHEMA.COLUMNS WHERE TABLE_NAME = {0}", tableName);
            using var reader = Database.CreateDataReader(cb);
            while (reader.Read()) cols.Add(reader.GetString(0));
            _tableColumns[tableName] = cols;
        }
    }
    ```

    ```csharp
    namespace DynamicWeb.Serializer.Configuration;

    /// <summary>
    /// Tokenizes a WHERE-clause string and validates it against a set of allowed column
    /// names. Rejects SQL injection patterns (`;`, `--`, `/*`, `EXEC`, `xp_`, subselects).
    /// Scope: single-table WHERE clauses with literal values. No joins, subqueries, or
    /// function calls.
    /// </summary>
    public class SqlWhereClauseValidator
    {
        private static readonly string[] BannedTokens = new[]
        {
            ";", "--", "/*", "*/", "xp_", "sp_executesql"
        };

        // Banned keywords (whole-word match, case-insensitive)
        private static readonly HashSet<string> BannedKeywords = new(StringComparer.OrdinalIgnoreCase)
        {
            "SELECT", "UPDATE", "DELETE", "INSERT", "MERGE", "EXEC", "EXECUTE", "DROP",
            "TRUNCATE", "ALTER", "CREATE", "GRANT", "REVOKE", "UNION", "INTO",
            "WAITFOR", "SHUTDOWN"
        };

        public void Validate(string whereClause, IReadOnlySet<string> allowedColumns)
        {
            if (string.IsNullOrWhiteSpace(whereClause))
                throw new InvalidOperationException("WHERE clause is empty.");

            // 1. Literal banned tokens (substring match, case-insensitive for alpha)
            foreach (var banned in BannedTokens)
                if (whereClause.Contains(banned, StringComparison.OrdinalIgnoreCase))
                    throw new InvalidOperationException(
                        $"WHERE clause contains banned token '{banned}': {whereClause}");

            // 2. Tokenize: strip single-quoted string literals first (they're safe),
            //    then split on whitespace + operators + parens.
            var stripped = StripStringLiterals(whereClause);
            var tokens = stripped.Split(new[] { ' ', '\t', '\n', '\r', '(', ')', ',', '=', '<', '>', '!' },
                                         StringSplitOptions.RemoveEmptyEntries);

            foreach (var token in tokens)
            {
                if (BannedKeywords.Contains(token))
                    throw new InvalidOperationException(
                        $"WHERE clause contains banned keyword '{token}': {whereClause}");

                // Identifier-like token (starts with a letter or underscore): must be in allowedColumns
                // or must be a safe keyword (AND, OR, NOT, IN, IS, NULL, LIKE, BETWEEN)
                if (IsIdentifierLike(token) && !IsSafeOperatorKeyword(token))
                {
                    if (!allowedColumns.Contains(token))
                        throw new InvalidOperationException(
                            $"WHERE clause references unknown identifier '{token}'. " +
                            $"Not a column on the target table. Check INFORMATION_SCHEMA. Clause: {whereClause}");
                }
            }
        }

        private static readonly HashSet<string> SafeOperatorKeywords =
            new(StringComparer.OrdinalIgnoreCase) { "AND", "OR", "NOT", "IN", "IS", "NULL", "LIKE", "BETWEEN", "TRUE", "FALSE" };
        private static bool IsSafeOperatorKeyword(string token) => SafeOperatorKeywords.Contains(token);
        private static bool IsIdentifierLike(string token) =>
            token.Length > 0 &&
            (char.IsLetter(token[0]) || token[0] == '_') &&
            !int.TryParse(token, out _);

        /// <summary>Replace single-quoted literal contents with placeholder (preserving structure).</summary>
        private static string StripStringLiterals(string input)
        {
            var sb = new System.Text.StringBuilder(input.Length);
            bool inString = false;
            foreach (var c in input)
            {
                if (c == '\'') { inString = !inString; sb.Append(c); continue; }
                if (inString) { sb.Append('x'); continue; }
                sb.Append(c);
            }
            if (inString)
                throw new InvalidOperationException("Unterminated string literal in WHERE clause.");
            return sb.ToString();
        }
    }
    ```

       ```csharp
       /// <summary>
       /// Curated list of runtime-only DW columns auto-excluded from SqlTable serialization.
       /// These columns hold visit counters, env-specific index pointers, and other data that
       /// is either recomputed at runtime or meaningful only in a specific environment.

           /// only when the column is demonstrably runtime-only on every DW 10.x install.

               new(StringComparer.OrdinalIgnoreCase)

               // F-07: visit counter on URL path table — recomputed at runtime, overwrites target on deploy

           };

           /// <summary>Returns the auto-excluded columns for a table (empty collection if none).</summary>
       ```
