Close the last two baseline-correctness gaps surfaced by the Swift 2.2 → CleanDB round-trip:

1. **TEMPLATE-01** (F-15) — The baseline references cshtml / grid-row json template files on disk, but templates are NOT shipped in the baseline (they deploy with code). Without a manifest, missing-template errors appear as runtime 500s after deserialize. Ship a manifest-only system: at serialize, record every template reference with its source page; at deserialize, validate each referenced file exists under `{filesRoot}/Templates/Designs/`. Under strict mode, missing templates are a hard failure (D-21).

2. **LINK-02 / F-07 / F-17** — Cross-environment page ID references. Two complementary passes:
   - **Pass 1 (serialize time, D-22):** Pre-commit sweep over the YAML tree. Every `Default.aspx?ID=N` reference is validated to resolve to a page whose `SourcePageId` is also in the baseline. Orphans and references-to-excluded-pages are caught before the baseline is committed to git. Serialize fails on unresolvable refs (treated like a correctness bug).
   - **Pass 2 (deserialize time, D-22):** Extend the existing `InternalLinkResolver` — which today only rewrites content-layer page fields — to also rewrite `Default.aspx?ID=N` patterns in configured SqlTable string columns (e.g. `UrlPath.UrlPathRedirect` from F-07). A new per-predicate `resolveLinksInColumns` field opts specific columns in. Unresolved refs at deserialize: WARNING lenient, FAIL strict (D-23).

**Out of scope:**
- Resolving non-`Default.aspx?ID=` patterns (raw numeric IDs in non-link contexts, ButtonEditor JSON `"SelectedValue"` — already covered by existing InternalLinkResolver) — extension to AreaId/CategoryId patterns is D-24 forward-work, not blocking
- Serializing template contents into the baseline (scope creep, explicitly deferred in D-19 / D-20)
- Pre-commit GitHub Action running the link sweep (v0.6.0 CI story per DEFERRED.md)

       using System.Text;
       using DynamicWeb.Serializer.Models;
       using YamlDotNet.Serialization;
       using YamlDotNet.Serialization.NamingConventions;

       namespace DynamicWeb.Serializer.Infrastructure;

       /// <summary>
       /// Per TEMPLATE-01 / D-19, D-20: manifest-only template tracking. Records
       /// which cshtml / grid-row / item-type files the baseline references; validates
       /// their presence at deserialize. Does NOT serialize template file contents.
       /// </summary>
       public record TemplateReference
       {
           public required string Path { get; init; }
           public required string Kind { get; init; }  // "page-layout" | "grid-row" | "item-type"
           public List<string> ReferencedBy { get; init; } = new();
       }

       public class TemplateAssetManifest
       {
           public const string ManifestFileName = "templates.manifest.yml";

           private readonly ISerializer _yamlWriter;
           private readonly IDeserializer _yamlReader;

           public TemplateAssetManifest()
           {
               _yamlWriter = new SerializerBuilder()
                   .WithNamingConvention(CamelCaseNamingConvention.Instance)
                   .Build();
               _yamlReader = new DeserializerBuilder()
                   .WithNamingConvention(CamelCaseNamingConvention.Instance)
                   .Build();
           }

           public void Write(string outputRoot, IEnumerable<TemplateReference> references)
           {
               Directory.CreateDirectory(outputRoot);
               var path = Path.Combine(outputRoot, ManifestFileName);
               var list = references
                   .OrderBy(r => r.Kind).ThenBy(r => r.Path, StringComparer.OrdinalIgnoreCase)
                   .ToList();
               File.WriteAllText(path, _yamlWriter.Serialize(list), Encoding.UTF8);
           }

           public List<TemplateReference>? Read(string inputRoot)
           {
               var path = Path.Combine(inputRoot, ManifestFileName);
               if (!File.Exists(path)) return null;
               return _yamlReader.Deserialize<List<TemplateReference>>(File.ReadAllText(path, Encoding.UTF8))
                      ?? new List<TemplateReference>();
           }

           /// <summary>
           /// For each reference, verify the file exists on the target filesystem. Missing
           /// references are escalated via the StrictModeEscalator. Per D-21: lenient =
           /// warn + continue; strict = escalate to CumulativeStrictModeException at end-of-run.
           /// </summary>
           public int Validate(
               string filesRoot,
               List<TemplateReference> references,
               Infrastructure.StrictModeEscalator escalator)
           {
               int missingCount = 0;
               var designsDir = Path.Combine(filesRoot, "Templates", "Designs");

               foreach (var r in references)
               {
                   bool found = r.Kind switch
                   {
                       "page-layout" => FindInAnyDesign(designsDir, r.Path),
                       "grid-row" => FindInAnyDesign(designsDir, Path.Combine("Grid", "Page", "RowDefinitions", $"{r.Path}.json")),
                       "item-type" => File.Exists(Path.Combine(filesRoot, "System", "Items", $"ItemType_{r.Path}.xml")),
                       _ => false
                   };

                   if (!found)
                   {
                       missingCount++;
                       var refList = r.ReferencedBy.Count > 5
                           ? string.Join(", ", r.ReferencedBy.Take(5)) + $", ... ({r.ReferencedBy.Count} total)"
                           : string.Join(", ", r.ReferencedBy);
                       escalator.Escalate($"Missing {r.Kind} template: '{r.Path}' — referenced by: {refList}");
                   }
               }

               return missingCount;
           }

           private static bool FindInAnyDesign(string designsDir, string relativePath)
           {
               if (!Directory.Exists(designsDir)) return false;
               return Directory.EnumerateDirectories(designsDir)
                   .Any(d => File.Exists(Path.Combine(d, relativePath)));
           }
       }

       using DynamicWeb.Serializer.Models;
       namespace DynamicWeb.Serializer.Infrastructure;

       /// <summary>
       /// Walks a list of SerializedPages and extracts TemplateReference records for
       /// every layout / item-type / grid-row reference. Accumulates "referencedBy"
       /// lists so validation failures can point the operator at source pages.
       /// </summary>
       public class TemplateReferenceScanner
       {
           public List<TemplateReference> Scan(List<SerializedPage> rootPages)
           {
               var byPath = new Dictionary<(string kind, string path), TemplateReference>();
               foreach (var p in rootPages) Walk(p, PagePath(p), byPath);
               return byPath.Values.ToList();
           }

           private static string PagePath(SerializedPage p) => p.UrlName ?? p.MenuText ?? p.PageUniqueId.ToString();

           private void Walk(SerializedPage page, string pageIdentifier, Dictionary<(string, string), TemplateReference> acc)
           {
               if (!string.IsNullOrEmpty(page.Layout))
                   AddRef(acc, "page-layout", page.Layout, pageIdentifier);

               if (!string.IsNullOrEmpty(page.ItemType))
                   AddRef(acc, "item-type", page.ItemType, pageIdentifier);

               foreach (var row in page.GridRows)
               {
                   // Check the SerializedGridRow DTO — its RowDefinition / DefinitionId property
                   // depends on the model shape; pick the field that holds the row-definition id
                   // (e.g. "1ColumnEmail"). If ambiguous, scan SerializedGridRow.cs first and adjust.
                   var rowDefId = GetRowDefinitionId(row);
                   if (!string.IsNullOrEmpty(rowDefId))
                       AddRef(acc, "grid-row", rowDefId, pageIdentifier);
               }

               foreach (var child in page.Children)
                   Walk(child, PagePath(child), acc);
           }

           private static string? GetRowDefinitionId(SerializedGridRow row)
           {
               // Best-effort extraction — match whichever field on SerializedGridRow holds the def id.
               // If no such field exists, return null and grid-row references won't be tracked
               // for this version of the model.
               // Inspect src/DynamicWeb.Serializer/Models/SerializedGridRow.cs before implementing.

    using System.Text.RegularExpressions;
    using DynamicWeb.Serializer.Models;

    namespace DynamicWeb.Serializer.Infrastructure;

    public record UnresolvedLink(string SourcePageIdentifier, string FieldName, int UnresolvablePageId, string Context);
    public record SweepResult(List<UnresolvedLink> Unresolved, int ResolvedCount);

    /// <summary>
    /// Per D-22 pass 1: scans every Default.aspx?ID=N reference in a serialized tree
    /// and verifies each target is ALSO in the tree (via SourcePageId). Emits a list
    /// of unresolvable references so the serializer can fail before commit.
    /// </summary>
    public class BaselineLinkSweeper
    {
        // Same patterns as InternalLinkResolver — reuse intent, not regex (kept local for decoupling)
        private static readonly Regex InternalLinkPattern = new(
            @"(Default\.aspx\?ID=)(\d+)(#(\d+))?",
            RegexOptions.IgnoreCase | RegexOptions.Compiled);
        private static readonly Regex SelectedValuePattern = new(
            @"(""SelectedValue"":\s*"")(\d+)("")",
            RegexOptions.Compiled);

        public SweepResult Sweep(List<SerializedPage> allPages)
        {
            var validSourceIds = new HashSet<int>();
            CollectSourceIds(allPages, validSourceIds);

            var unresolved = new List<UnresolvedLink>();
            int resolved = 0;

            foreach (var page in allPages)
                WalkPage(page, PageIdent(page), validSourceIds, unresolved, ref resolved);

            return new SweepResult(unresolved, resolved);
        }

        private static string PageIdent(SerializedPage p) =>
            $"page {p.PageUniqueId} ({p.UrlName ?? p.MenuText ?? "unnamed"})";

        private static void CollectSourceIds(IEnumerable<SerializedPage> pages, HashSet<int> acc)
        {
            foreach (var p in pages)
            {
                if (p.SourcePageId.HasValue) acc.Add(p.SourcePageId.Value);
                CollectSourceIds(p.Children, acc);
            }
        }

        private void WalkPage(
            SerializedPage page, string ident, HashSet<int> validIds,
            List<UnresolvedLink> unresolved, ref int resolved)
        {
            CheckField(page.ShortCut, ident, "ShortCut", validIds, unresolved, ref resolved);

            if (page.NavigationSettings != null)
                CheckField(page.NavigationSettings.ProductPage, ident, "NavigationSettings.ProductPage",
                    validIds, unresolved, ref resolved);

            foreach (var kvp in page.Fields ?? new Dictionary<string, object>())
                if (kvp.Value is string s)
                    CheckField(s, ident, $"Fields.{kvp.Key}", validIds, unresolved, ref resolved);

            foreach (var kvp in page.PropertyFields ?? new Dictionary<string, object>())
                if (kvp.Value is string s)
                    CheckField(s, ident, $"PropertyFields.{kvp.Key}", validIds, unresolved, ref resolved);

            // Walk paragraphs
            foreach (var row in page.GridRows)
                foreach (var col in row.Columns)
                    foreach (var para in col.Paragraphs)
                    {
                        var paraIdent = $"{ident}/paragraph {para.ParagraphUniqueId}";
                        foreach (var kvp in para.Fields ?? new Dictionary<string, object>())
                            if (kvp.Value is string s)
                                CheckField(s, paraIdent, $"Fields.{kvp.Key}", validIds, unresolved, ref resolved);
                    }

            foreach (var c in page.Children)
                WalkPage(c, PageIdent(c), validIds, unresolved, ref resolved);
        }

        private static void CheckField(
            string? value, string sourceIdent, string fieldName,
            HashSet<int> validIds, List<UnresolvedLink> unresolved, ref int resolved)
        {
            if (string.IsNullOrEmpty(value)) return;

            // Internal Default.aspx?ID= references
            foreach (Match m in InternalLinkPattern.Matches(value))
            {
                if (int.TryParse(m.Groups[2].Value, out var id))
                {
                    if (validIds.Contains(id)) { resolved++; continue; }
                    unresolved.Add(new UnresolvedLink(sourceIdent, fieldName, id, m.Value));
                }
            }

            // ButtonEditor "SelectedValue" JSON
            foreach (Match m in SelectedValuePattern.Matches(value))
            {
                if (int.TryParse(m.Groups[2].Value, out var id))
                {
                    if (validIds.Contains(id)) { resolved++; continue; }
                    unresolved.Add(new UnresolvedLink(sourceIdent, fieldName, id, m.Value));
                }
            }

            // Raw numeric (LinkEditor): only flag if the entire trimmed string is a pure number
            // AND matches an ID that's NOT in the valid set but IS suspicious (heuristic — skip for
            // now; false-positive risk is too high). Document limitation.
        }
    }

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| YAML template references → filesystem path validation | Path strings from the manifest are read-only; Validate joins them with `filesRoot` via `Path.Combine` — could be tricked by `../` traversal if not guarded |
| SqlTable string values being rewritten pre-MERGE | The rewritten value goes to a parameterized MERGE via CommandBuilder — no SQL injection path. But the REWRITE itself could alter meaning if the regex is manipulated through malicious source data. |
| BaselineLinkSweeper exception message | Contains page GUIDs + field names from the source baseline — may include PII if the baseline includes customer content |
