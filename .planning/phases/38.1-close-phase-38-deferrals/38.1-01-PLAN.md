// File: src/DynamicWeb.Serializer/Infrastructure/ParagraphIdCollector.cs
namespace DynamicWeb.Serializer.Infrastructure;

internal static class ParagraphIdCollector
{
    // Visits every SerializedParagraph in the tree (pages → GridRows →
    // Columns → Paragraphs, recursing through Children) and invokes the
    // supplied visitor for each one. Shared by BaselineLinkSweeper
    // (accumulates a HashSet of SourceParagraphId) and InternalLinkResolver
    // (builds a source→target Dictionary using a paragraph GUID cache).
    public static void Visit(
        IEnumerable<SerializedPage> pages,
        Action<SerializedParagraph> visitor)
    {
        foreach (var p in pages)
        {
            foreach (var row in p.GridRows)
                foreach (var col in row.Columns)
                    foreach (var para in col.Paragraphs)
                        visitor(para);
            Visit(p.Children, visitor);
        }
    }
}
