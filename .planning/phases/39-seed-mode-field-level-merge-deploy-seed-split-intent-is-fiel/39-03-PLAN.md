    Step 3 — Add `Invoke-Sqlcmd-StringScalar` helper near the existing `Invoke-Sqlcmd-Scalar` definition. Concrete signature:

    ```powershell
    function Invoke-Sqlcmd-StringScalar {
        param(
            [Parameter(Mandatory=$true)][string]$Server,
            [Parameter(Mandatory=$true)][string]$Database,
            [Parameter(Mandatory=$true)][string]$Query,
            [int]$MaxCharsReturned = 100000
        )
        # -h -1: suppress headers
        # -W: trim trailing whitespace per row
        # -s '|' : column separator (we only expect one column)
        # -y 0 -Y 0: unlimited character length (within MaxCharsReturned guard below)
        $raw = & sqlcmd -S $Server -E -d $Database -h -1 -W -s '|' -y 0 -Y 0 -Q "SET NOCOUNT ON; $Query" 2>&1
        if ($LASTEXITCODE -ne 0) {
            throw "sqlcmd failed on $Database: $raw"
        }
        $nonBlank = $raw | Where-Object { $_ -and $_.Trim().Length -gt 0 }
        $line = $nonBlank | Select-Object -First 1
        if (-not $line) { return $null }
        $s = $line.ToString().Trim()
        if ($s.Length -gt $MaxCharsReturned) {
            $s = $s.Substring(0, $MaxCharsReturned)
        }
        return $s
    }
    ```

    ```powershell
    if ($Mode -eq 'DeployThenTweakThenSeed') {
        Write-Host ''
        Write-Host '============================================================'
        Write-Host 'Phase 39 D-15: Deploy -> Tweak -> Seed sub-pipeline'
        Write-Host '============================================================'

        # Step 15.0 — Preflight: capture baseline state for assertions
        Write-Host '  [15.0] Preflight: capturing pre-tweak state'
        $preTweakValue = Invoke-Sqlcmd-StringScalar `
            -Server $SqlServer -Database $CleanDb `
            -Query "SELECT $TweakColumnName FROM Page WHERE ID=$TweakPageId"
        Write-Host "    Pre-tweak [$TweakColumnName]@Page.$TweakPageId = '$preTweakValue'"

        # Step 15.1 — Inject customer tweak between Deploy and Seed
        Write-Host '  [15.1] Injecting customer tweak'
        & sqlcmd -S $SqlServer -E -d $CleanDb -Q @"
SET NOCOUNT ON;
UPDATE Page SET $TweakColumnName = N'$TweakValue' WHERE ID = $TweakPageId;
"@
        if ($LASTEXITCODE -ne 0) { throw "Failed to inject customer tweak" }

        $postTweakValue = Invoke-Sqlcmd-StringScalar `
            -Server $SqlServer -Database $CleanDb `
            -Query "SELECT $TweakColumnName FROM Page WHERE ID=$TweakPageId"
        if ($postTweakValue -ne $TweakValue) {
            throw "Tweak injection failed — expected '$TweakValue', got '$postTweakValue'"
        }

        # Step 15.2 — Run Seed deserialize
        Write-Host '  [15.2] Deserialize Seed (first pass)'
        $desSeed1 = Invoke-DwApi -HostUrl $CleanDbHostUrl `
            -Endpoint '/Admin/Api/SerializerDeserialize?mode=seed' `
            -LogPath (Join-Path $script:runDir 'deserialize-seed-pass1.log')
        if ($desSeed1.Code -ne 200) {
            throw "Deserialize Seed pass 1: expected HTTP 200, got $($desSeed1.Code)"
        }
        if (Select-String -Path (Join-Path $script:runDir 'deserialize-seed-pass1.log') `
                          -Pattern 'escalated|CumulativeStrictModeException' -Quiet) {
            throw "Deserialize Seed pass 1 emitted strict-mode escalations"
        }
        # D-11 log format regression guard
        if (Select-String -Path (Join-Path $script:runDir 'deserialize-seed-pass1.log') `
                          -Pattern 'Seed-skip:' -Quiet) {
            throw "Deserialize Seed pass 1 emitted 'Seed-skip:' — Phase 39 D-11 regression"
        }
        if (-not (Select-String -Path (Join-Path $script:runDir 'deserialize-seed-pass1.log') `
                                -Pattern 'Seed-merge:' -Quiet)) {
            throw "Deserialize Seed pass 1 emitted NO 'Seed-merge:' lines — merge branch not firing"
        }

        # Step 15.3 — Assert customer tweak is preserved
        Write-Host '  [15.3] Verifying customer tweak preserved'
        $postSeedTweak = Invoke-Sqlcmd-StringScalar `
            -Server $SqlServer -Database $CleanDb `
            -Query "SELECT $TweakColumnName FROM Page WHERE ID=$TweakPageId"
        if ($postSeedTweak -ne $TweakValue) {
            throw "Customer tweak NOT preserved — expected '$TweakValue', got '$postSeedTweak'. Seed merge branch overwrote a set field."
        }
        Write-Host "    [PASS] Customer tweak preserved: '$postSeedTweak'"

        # Step 15.4 — Assert Mail1SenderEmail filled in EcomPayments.PaymentGatewayParameters
        Write-Host '  [15.4] Verifying Mail1SenderEmail filled in EcomPayments XML'
        $paymentXml = Invoke-Sqlcmd-StringScalar `
            -Server $SqlServer -Database $CleanDb `
            -Query "SELECT CAST(PaymentGatewayParameters AS NVARCHAR(MAX)) FROM EcomPayments WHERE PaymentID = N'$PaymentRowPaymentId'"
        if (-not $paymentXml) {
            throw "EcomPayments row '$PaymentRowPaymentId' not found on CleanDB"
        }
        if ($paymentXml -notmatch 'Mail1SenderEmail') {
            throw "Mail1SenderEmail NOT filled by Seed XML-merge (D-15 acceptance failed). XML = $paymentXml"
        }
        Write-Host '    [PASS] Mail1SenderEmail present in PaymentGatewayParameters'

        # Step 15.5 — Second-pass Seed: idempotency (D-09)
        Write-Host '  [15.5] Deserialize Seed (second pass — idempotency check)'
        $desSeed2 = Invoke-DwApi -HostUrl $CleanDbHostUrl `
            -Endpoint '/Admin/Api/SerializerDeserialize?mode=seed' `
            -LogPath (Join-Path $script:runDir 'deserialize-seed-pass2.log')
        if ($desSeed2.Code -ne 200) {
            throw "Deserialize Seed pass 2: expected HTTP 200, got $($desSeed2.Code)"
        }
        # Second-pass EVERY Seed-merge line should show 0 filled
        $pass2Lines = Get-Content (Join-Path $script:runDir 'deserialize-seed-pass2.log')
        $nonZeroFills = $pass2Lines | Where-Object { $_ -match 'Seed-merge:.* - (\d+) filled' -and [int]$matches[1] -gt 0 }
        if ($nonZeroFills) {
            Write-Warning "D-09 idempotency soft warn: Pass 2 Seed-merge lines with non-zero fills:"
            $nonZeroFills | ForEach-Object { Write-Warning "    $_" }
            throw "D-09 idempotency failed — second Seed pass wrote new fields. See deserialize-seed-pass2.log"
        }
        Write-Host '    [PASS] D-09 idempotency — second Seed pass wrote zero fields'

        # Step 15.6 — Assert tweak still preserved after second pass
        $finalTweak = Invoke-Sqlcmd-StringScalar `
            -Server $SqlServer -Database $CleanDb `
            -Query "SELECT $TweakColumnName FROM Page WHERE ID=$TweakPageId"
        if ($finalTweak -ne $TweakValue) {
            throw "After second Seed pass, tweak value lost — expected '$TweakValue', got '$finalTweak'"
        }

        Write-Host ''
        Write-Host '[ALL PASS] Phase 39 D-15 live acceptance gate closed.'
    }
    ```
