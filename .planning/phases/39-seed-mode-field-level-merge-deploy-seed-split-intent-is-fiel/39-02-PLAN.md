## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| YAML row dict -> SqlTableWriter.UpdateColumnSubset | YAML column values flow into parameterized SQL. Identical to existing source-wins MERGE path — no new injection surface. |
| YAML XML column value -> XmlMergeHelper -> SQL column write | XML parsed via XDocument; malformed XML defensively returns target (no throw). Merged XML serialized and stored as column value; same column-level write surface as today. |
| SqlTableProvider -> DW target database | Writes run under the DW DB connection; identity whitelist (Phase 37-03 SqlIdentifierValidator) already governs table/column names. |

    ```csharp
    /// <summary>
    /// Phase 39 D-17: issues a targeted UPDATE statement writing only the requested
    /// column subset, scoped by the key-column identity predicate. Used by
    /// SqlTableProvider's Seed-merge branch after per-column merge planning has
    /// determined which target columns are "unset" per MergePredicate.
    /// </summary>
    /// <param name="tableName">Target table — identifier already whitelisted at config-load by Phase 37-03 SqlIdentifierValidator.</param>
    /// <param name="keyColumns">Key columns forming the WHERE identity predicate.</param>
    /// <param name="fullRow">Complete YAML row — identity columns read from here plus any column in columnsToUpdate.</param>
    /// <param name="columnsToUpdate">The subset of columns to write (pre-filtered by caller per merge predicate).</param>
    /// <param name="isDryRun">When true, logs the would-be SQL and returns Updated without executing.</param>
    /// <param name="log">Optional log sink for the dry-run trace line.</param>
    /// <remarks>
    /// Dry-run logs include actual values — see 39-CONTEXT.md D-19 / threat T-39-02-06.
    /// Not intended for log sinks shared with untrusted parties.
    /// </remarks>
    public virtual WriteOutcome UpdateColumnSubset(
        string tableName,
        IReadOnlyList<string> keyColumns,
        Dictionary<string, object?> fullRow,
        IEnumerable<string> columnsToUpdate,
        bool isDryRun,
        Action<string>? log = null)
    {
        var colList = columnsToUpdate.ToList();
        if (colList.Count == 0)
        {
            // Caller pre-filtered to empty — no-op. Return Updated so caller's counter semantics stay consistent.
            return WriteOutcome.Updated;
        }

        try
        {
            var cb = new CommandBuilder();
            cb.Add($"UPDATE [{tableName}] SET ");

            for (int i = 0; i < colList.Count; i++)
            {
                if (i > 0) cb.Add(",");
                var col = colList[i];
                var val = fullRow.TryGetValue(col, out var v) ? v ?? DBNull.Value : DBNull.Value;
                cb.Add($"[{col}]=");
                cb.Add("{0}", val);
            }

            cb.Add(" WHERE ");
            for (int i = 0; i < keyColumns.Count; i++)
            {
                if (i > 0) cb.Add(" AND ");
                var keyCol = keyColumns[i];
                var keyVal = fullRow.TryGetValue(keyCol, out var kv) ? kv ?? DBNull.Value : DBNull.Value;
                cb.Add($"[{keyCol}]=");
                cb.Add("{0}", keyVal);
            }

            if (isDryRun)
            {
                log?.Invoke($"    [DRY-RUN] UPDATE [{tableName}] SET {string.Join(",", colList.Select(c => $"[{c}]=?"))} WHERE {string.Join(" AND ", keyColumns.Select(k => $"[{k}]=?"))}");
                return WriteOutcome.Updated;
            }

            _sqlExecutor.ExecuteNonQuery(cb);
            return WriteOutcome.Updated;
        }
        catch (Exception ex)
        {
            log?.Invoke($"    ERROR [{tableName}].UpdateColumnSubset: {ex.Message}");
            return WriteOutcome.Failed;
        }
    }
    ```

    using System;
    using System.Collections.Generic;
    using System.IO;
    using System.Linq;
    using System.Text;
    using System.Xml;
    using System.Xml.Linq;

    namespace DynamicWeb.Serializer.Infrastructure;

    /// <summary>
    /// Per-element XML merge for Phase 39 D-22..D-27. Fills target XML elements
    /// that are absent or have null/empty/whitespace text, using source XML as
    /// the fill value pool. Preserves any element present on target but absent
    /// from source (D-24 — Seed never strips).
    /// </summary>
    /// <remarks>
    /// Element identity is by local-name, except when a child has a <c>name</c>
    /// attribute (DW <c>&lt;Parameter name="X"&gt;</c> idiom used by
    /// EcomPayments.PaymentGatewayParameters and EcomShippings.ShippingServiceParameters),
    /// in which case the name attribute is the identity key. This matches the
    /// identity scheme already used by <see cref="XmlFormatter.CompactWithMerge"/>.
    ///
    /// Security (T-39-02-05): uses an XmlReader with DtdProcessing=Prohibit to
    /// reject entity-expansion (billion-laughs) payloads. Malformed XML on either
    /// side returns the target unchanged.
    ///
    /// Dry-run (T-39-02-06): the MergeWithDiagnostics variant returns per-element
    /// fill descriptions that include source values — do not emit those logs to
    /// untrusted log sinks.
    /// </remarks>
    public static class XmlMergeHelper
    {
        private static readonly XmlReaderSettings SafeSettings = new()
        {
            DtdProcessing = DtdProcessing.Prohibit,
            XmlResolver = null,
            IgnoreComments = false,
            IgnoreWhitespace = false
        };

        public static string? Merge(string? targetXml, string? sourceXml)
            => MergeWithDiagnostics(targetXml, sourceXml).mergedXml;

        public static (string? mergedXml, IReadOnlyList<string> fills) MergeWithDiagnostics(string? targetXml, string? sourceXml)
        {
            var fills = new List<string>();

            if (string.IsNullOrWhiteSpace(targetXml) && string.IsNullOrWhiteSpace(sourceXml))
                return (null, fills);
            if (string.IsNullOrWhiteSpace(targetXml))
                return (sourceXml, fills);  // nothing to merge against — source becomes merged
            if (string.IsNullOrWhiteSpace(sourceXml))
                return (targetXml, fills);

            XDocument? targetDoc = TryParse(targetXml);
            XDocument? sourceDoc = TryParse(sourceXml);

            if (targetDoc?.Root == null || sourceDoc?.Root == null)
                return (targetXml, fills);  // defensive — malformed either side, return target unchanged

            MergeElement(targetDoc.Root, sourceDoc.Root, fills);

            var writerSettings = new XmlWriterSettings { OmitXmlDeclaration = !targetXml.Contains("<?xml"), Indent = false };
            using var sw = new StringWriter();
            using (var xw = XmlWriter.Create(sw, writerSettings))
                targetDoc.Save(xw);
            return (sw.ToString(), fills);
        }

        private static XDocument? TryParse(string xml)
        {
            try
            {
                using var sr = new StringReader(xml);
                using var xr = XmlReader.Create(sr, SafeSettings);
                return XDocument.Load(xr);
            }
            catch (XmlException) { return null; }
        }

        private static string GetKey(XElement el)
            => el.Attribute("name")?.Value ?? el.Name.LocalName;

        private static bool IsUnsetText(string? text)
            => string.IsNullOrWhiteSpace(text);

        private static bool IsUnsetElement(XElement el)
            => !el.HasElements && IsUnsetText(el.Value);

        private static void MergeElement(XElement target, XElement source, List<string> fills)
        {
            // 1. Leaf-text fill: if target is a leaf and source has a non-empty text, fill.
            if (!target.HasElements && !source.HasElements)
            {
                if (IsUnsetText(target.Value) && !IsUnsetText(source.Value))
                {
                    fills.Add($"element={GetKey(target)}: <empty> -> '{source.Value}'");
                    target.Value = source.Value;
                }
                // Attribute-level merge on this leaf
                MergeAttributes(target, source, fills);
                return;
            }

            // 2. Attribute-level merge on this level.
            MergeAttributes(target, source, fills);

            // 3. Index target children by identity key.
            var targetByKey = target.Elements()
                .GroupBy(GetKey, StringComparer.OrdinalIgnoreCase)
                .ToDictionary(g => g.Key, g => g.First(), StringComparer.OrdinalIgnoreCase);

            foreach (var srcChild in source.Elements())
            {
                var key = GetKey(srcChild);
                if (targetByKey.TryGetValue(key, out var tgtChild))
                {
                    if (IsUnsetElement(tgtChild))
                    {
                        // Element present on target but empty/whitespace — replace with source content.
                        fills.Add($"element={key}: <missing-or-empty> -> '{srcChild.Value}'");
                        tgtChild.ReplaceWith(new XElement(srcChild));
                    }
                    else
                    {
                        // Recurse — per-property merge at deeper level.
                        MergeElement(tgtChild, srcChild, fills);
                    }
                }
                else
                {
                    // Element absent on target — add it. D-22.
                    fills.Add($"element={key}: <missing> -> '{srcChild.Value}'");
                    target.Add(new XElement(srcChild));
                }
            }
            // D-24: target-only elements preserved by doing nothing for them.
        }

        private static void MergeAttributes(XElement target, XElement source, List<string> fills)
        {
            foreach (var srcAttr in source.Attributes())
            {
                var tgtAttr = target.Attribute(srcAttr.Name);
                if (tgtAttr == null)
                {
                    fills.Add($"attribute={srcAttr.Name}: <missing> -> '{srcAttr.Value}'");
                    target.SetAttributeValue(srcAttr.Name, srcAttr.Value);
                }
                else if (IsUnsetText(tgtAttr.Value) && !IsUnsetText(srcAttr.Value))
                {
                    fills.Add($"attribute={srcAttr.Name}: <empty> -> '{srcAttr.Value}'");
                    tgtAttr.Value = srcAttr.Value;
                }
            }
        }
    }
